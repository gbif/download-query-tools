22  /**
23   * Guice module providing a TitleLookup instance for the HumanFilterBuilder to lookup species and dataset titles.
24   */
25  public class TitleLookupModule extends PrivateModule {
26    private static final Logger LOG = LoggerFactory.getLogger(TitleLookupModule.class);
27    private final boolean provideHttpClient;
28    private final URI apiRoot;
29  
30    /**
31     * @param provideHttpClient if true the module creates a new internal only http client instance
32     */
33    public TitleLookupModule(boolean provideHttpClient, String apiRoot) {
34      this.provideHttpClient = provideHttpClient;
35      this.apiRoot = URI.create(Preconditions.checkNotNull(apiRoot, "API url can't be null"));
36      Preconditions.checkArgument(this.apiRoot.getHost()!=null, "API url must have a host! " + apiRoot);
37      LOG.info("Create new TitleLookup using API " + apiRoot);
38    }
39  
40    @Override
41    protected void configure() {
42      if (provideHttpClient) {
43        bind(HttpClient.class).toInstance(provideHttpClient());
44      }
45      expose(TitleLookup.class);
46    }
47  
48    @Provides
49    @Singleton
50    @Inject
51    public TitleLookup provideLookup(HttpClient hc) {
52      ApacheHttpClient4Handler hch = new ApacheHttpClient4Handler(hc, null, false);
53      ClientConfig clientConfig = new DefaultClientConfig();
54      clientConfig.getFeatures().put(JSONConfiguration.FEATURE_POJO_MAPPING, Boolean.TRUE);
55      WebResource api = new ApacheHttpClient4(hch, clientConfig).resource(apiRoot);
56      return new TitleLookup(api);
57    }
58  
59    private HttpClient provideHttpClient() {
60      return HttpUtil.newMultithreadedClient(5000, 10, 10);
61    }
62  
63  }
