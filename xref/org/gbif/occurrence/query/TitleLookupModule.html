44      ClientConfig clientConfig = new DefaultClientConfig();
45      clientConfig.getFeatures().put(JSONConfiguration.FEATURE_POJO_MAPPING, Boolean.TRUE);
46      WebResource api = new ApacheHttpClient4(hch, clientConfig).resource(apiRoot);
47      return new TitleLookup(api);
