20    public void testLookup() throws Exception {

22      when(res.path(anyString())).thenReturn(res);
23  
24      NameUsage abies = new NameUsage();
25      abies.setScientificName("Abies alba Mill.");
26      when(res.get(NameUsage.class)).thenReturn(abies);
27  
28      Dataset pontaurus = new Dataset();
29      pontaurus.setTitle("PonTaurus");
30      when(res.get(Dataset.class)).thenReturn(pontaurus);
31  
32      TitleLookup tl = new TitleLookup(res);
33  
34      assertEquals("Abies alba Mill.", tl.getSpeciesName("4231"));
35      assertEquals("PonTaurus", tl.getDatasetTitle(UUID.randomUUID().toString()));
36    }
