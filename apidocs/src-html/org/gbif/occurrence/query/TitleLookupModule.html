001package org.gbif.occurrence.query;
002
003import org.gbif.utils.HttpUtil;
004
005import com.google.inject.AbstractModule;
006import com.google.inject.Inject;
007import com.google.inject.Provides;
008import com.google.inject.Singleton;
009import com.sun.jersey.api.client.WebResource;
010import com.sun.jersey.api.client.config.ClientConfig;
011import com.sun.jersey.api.client.config.DefaultClientConfig;
012import com.sun.jersey.api.json.JSONConfiguration;
013import com.sun.jersey.client.apache4.ApacheHttpClient4;
014import com.sun.jersey.client.apache4.ApacheHttpClient4Handler;
015import org.apache.http.client.HttpClient;
016
017/**
018 * Guice module providing a TitleLookup instance for the HumanFilterBuilder to lookup species and dataset titles.
019 */
020public class TitleLookupModule extends AbstractModule {
021  private final boolean provideHttpClient;
022  private final String apiRoot;
023
024  /**
025   * @param provideHttpClient if true the module creates a new default http client instance
026   */
027  public TitleLookupModule(boolean provideHttpClient, String apiRoot) {
028    this.provideHttpClient = provideHttpClient;
029    this.apiRoot = apiRoot;
030  }
031
032  @Override
033  protected void configure() {
034    if (provideHttpClient) {
035      bind(HttpClient.class).toInstance(provideHttpClient());
036    }
037  }
038
039  @Provides
040  @Singleton
041  @Inject
042  private TitleLookup provideLookup(HttpClient hc) {
043    ApacheHttpClient4Handler hch = new ApacheHttpClient4Handler(hc, null, false);
044    ClientConfig clientConfig = new DefaultClientConfig();
045    clientConfig.getFeatures().put(JSONConfiguration.FEATURE_POJO_MAPPING, Boolean.TRUE);
046    WebResource api = new ApacheHttpClient4(hch, clientConfig).resource(apiRoot);
047    return new TitleLookup(api);
048  }
049
050  private HttpClient provideHttpClient() {
051    return HttpUtil.newMultithreadedClient(5000, 10, 10);
052  }
053}
